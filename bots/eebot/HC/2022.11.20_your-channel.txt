		self._ws.send(
			json.dumps(

				[
					self.csvdata,

				],

			self.updatefile()

```python
import json
import time
import random
import pandas as pd
from websocket import create_connection

class Client:
	def __init__(self):
		self._ws = create_connection("wss://hack.chat/chat-ws")
		if self._ws.status == 101:
			print("Able to run.")
			self.csvdata = self.loadfile()
			self.record()

	def loadfile(self):
		try:
			return pd.read_csv("data.csv")
		except pd.errors.EmptyDataError:
			return pd.DataFrame(
				[]
			)

	def updatefile(self):
		self.csvdata.to_csv("data.csv")

	def record(self):
		self._ws.send(
			json.dumps(
				{
					"cmd":"join",
					"nick":f"light{random.randint(999,10000)}",
					"channel":"your-channel"
				}
			)
		)
		while True:
			jsondata = json.loads(self._ws.recv())
			cmd = jsondata["cmd"]
			if cmd != "chat":
				continue
			rv = pd.Series(
				jsondata,
				index=["time","nick","trip","uType","text"]
			)
			rv["time"] = time.strftime(
				"%Y-%m-%d %H:%M:%S",
				time.localtime(rv["time"]/1000)
			)
			print(rv.to_string())
			self.csvdata = pd.concat(
				[
					self.csvdata,
					rv.to_frame().T
				],
				ignore_index=True
			)
			self.updatefile()

if __name__ == '__main__':
	Client()
```

def lookup():
   reply = ""
   if request.method == "GET":
      timearea = definearg("time",request.args) # 2020-5-12/2020-5-13
      if timearea != None:
         try:
            start,end = [
               [int(j) for j in i.split("-")]
               for i in timearea.split("/")
            ] # [[2020,5,12],[2020,5,13]]
         except:
            timearea = None

      for s in pd.read_csv("data.csv",index_col=0).itertuples():

               reply = reply + f"{str(s)}<br>"

            elif start[1]<=stime[1]<=end[1]:
               reply = reply + f"{str(s)}<br>"
               continue
            elif start[2]<=stime[2]<=end[2]:
               reply = reply + f"{str(s)}<br>"

```
    with open("index","w+") as f:
        f.write(generate_index(filename,info))
    os.rename(filename,"FILE")
    with open("FILE","rb") as f:
        bindata=f.read()
    en=Fernet(password)

    os.rename("data.zip",filename+".szip")

def ziptofile(filename,password):
    with zipfile.ZipFile(filename,"r") as z:
        z.extract("FILE",".")
        z.extract("index",".")
    with open("index","r") as f:
        Filename=json.loads(f.read())["index_name"]
    with open("FILE","rb") as f:
        data=f.read()
    de=Fernet(keyconvert(password))
    with open("FILE","wb+") as f:
        f.write(de.decrypt(data))
    os.rename("FILE",Filename)
    os.remove("index")
def keyconvert(key):
    if type(key) == str:
```

1，遇到不小心把水倒到笔记本键盘上时，不要惊慌失措，记着第一反应不是去关电脑开关而是拔电源插头，关电脑开关关机是要时间的。

2，应立即把笔记本倒放，让水向下不会再往主板方向流了，记着在没有确认干的情况下别正放。

3，现在最要紧的是赶快找吸水性很好的纸或其他东西，我觉得卫生纸最好又不贵，扯一堆，有电脑的宽度，放在桌子边沿，在把朝下的笔记本进水键盘方向放在纸上4，纸要翻动，让干的纸继续吸5，你觉得干了，笔记本不要正翻倒着用电风吹慢慢吹，这是个细活不能急，觉得OK了，在放正，朝电脑键盘急空隙的地方吹6，总之，就是要把它给吹干，7，一般情况下键盘都能用，就是触摸板会失灵出问题，这是因为触摸板里面集成许多灵敏的线路器件，只要一点点没干都会出现失灵，及点击程序出错，8，吹干如果能开机，（有些时候开不了说明还没干，电脑自我保护，在吹，放在通风的地方放段时间），触摸触摸板指针能动，左右键时而能用时而不能用，一般来说不是驱动的问题，硬件也还是好的，这就需要多用电风吹吹吹，电脑时常开着，会产生热量把水分蒸发，过个三四天，应该就好了。

9，如果一直开不了机，或能开机毛病不能消除，那只能送去维修了。10，小小提议，维修最好是自己现场监督，因为我觉得不是很信得过。维修时一定要记住帮你维修的所谓的工程师的姓明，维修证明等，还有修好了第一时间要查看自己电脑配置，与原机配置是否符合，等等....... 希望对你有用。
