command = ["11->C:C+1->A","A+1->A"]

runtimes = 1
def casio_pro(commands):
    global runtimes
    import os
    # If the file is not exist,create a new one
    if not os.path.exists("CPU.casio"):
        with open("CPU.casio","w") as f:
            f.write("0\n"*8)
    # Split the command string to normal python data type
    command = [c.strip() for c in commands[runtimes-1].split(":")]
    com_command = []
    is_syntax = 0
    for c in command:

                is_syntax += 1
                if is_syntax == 2:
                    raise "Syntax Error."
                com_command.append((c[:i],c[i+2:]))
        is_syntax = 0
    max_iter = 100000
    if runtimes == 1:
        max_iter = 1
    for j in range(max_iter):
        for val,var in com_command:
            # Update the data

    if runtimes == 1:

        casio_pro(commands)
    elif runtimes == 2:
